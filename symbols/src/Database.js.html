1 import { createRequire } from "module";
  2 const require = createRequire(import.meta.url);
  3 import pkg from "pg";
  4 const { Client } = pkg;
  5 require("dotenv").config();
  6 
  7 class Database {
  8   constructor() {
  9     this.client = new Client({
 10       user: process.env.USER_CLIENT,
 11       host: process.env.HOST,
 12       database: process.env.DATABASE,
 13       password: process.env.PASSWORD,
 14       port: process.env.PORT,
 15     });
 16     this.connected = false;
 17   }
 18 
 19   async connect() {
 20     try {
 21       if (!this.connected) {
 22         await this.client.connect();
 23         this.connected = true;
 24         console.log("Connected to database successfully");
 25       }
 26     } catch (err) {
 27       console.error("Connection error", err.stack);
 28     }
 29   }
 30 
 31   async query(sql, params) {
 32     try {
 33       const result = await this.client.query(sql, params);
 34       return result.rows;
 35     } catch (err) {
 36       console.error("Error executing query", err.stack);
 37     }
 38   }
 39 
 40   async insert(sql, params) {
 41     try {
 42       await this.client.query(sql, params);
 43     } catch (err) {
 44       console.error("Error executing insert", err.stack);
 45     }
 46   }
 47 
 48   async delete(sql, params) {
 49     try {
 50       await this.client.query(sql, params);
 51     } catch (err) {
 52       console.error("Error executing delete", err.stack);
 53     }
 54   }
 55 
 56   async disconnect() {
 57     try {
 58       if (this.connected) {
 59         await this.client.end();
 60         this.connected = false;
 61         console.log("Disconnected from database");
 62       }
 63     } catch (err) {
 64       console.error("Error disconnecting from database", err.stack);
 65     }
 66   }
 67 }
 68 
 69 export { Database };
