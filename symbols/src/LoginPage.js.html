311 const apiKey = `${process.env.WEATHER_API_KEY}`;
312 // Render the weather api for guest checkout
313 function guestRenderWeather(req, res, page) {
314   // Get city name passed in the form
315   let city = req.body.city;
316 
317   // Use that city name to fetch data
318   // Use the API_KEY in the '.env' file
319   let url = `http://api.openweathermap.org/data/2.5/weather?q=${city}&units=metric&appid=${apiKey}`;
320 
321   fetch(url)
322     .then((res) => res.json())
323     .then(async (weather) => {
324       const entree = new Menu();
325       const drink = new Menu();
326       const salad = new Menu();
327       const side = new Menu();
328       const treat = new Menu();
329 
330       const entreeItems = await entree.displayEntrees();
331       const drinkItems = await drink.displayDrinks();
332       const saladItems = await salad.displaySalads();
333       const sideItems = await side.displaySides();
334       const treatItems = await treat.displayTreats();
335       if (weather.main == undefined) {
336         res.render(page, {
337           userRole,
338           treatItems,
339           entreeItems,
340           drinkItems,
341           sideItems,
342           saladItems,
343           weather: null,
344           error: "Error, please try again",
345         });
346       } else {
347         let place = `${weather.name}, ${weather.sys.country}`,
348           weatherTimezone = `${new Date(
349             weather.dt * 1000 - weather.timezone * 1000
350           )}`;
351         let weatherTemp = `${weather.main.temp}`,
352           weatherIcon = `http://openweathermap.org/img/wn/${weather.weather[0].icon}@2x.png`,
353           main = `${weather.weather[0].main}`,
354           weatherFahrenheit;
355         weatherFahrenheit = (weatherTemp * 9) / 5 + 32;
356 
357         function roundToTwo(num) {
358           return +(Math.round(num + "e+2") + "e-2");
359         }
360         weatherFahrenheit = roundToTwo(weatherFahrenheit);
361 
362         userRole = userRoles[2].role;
363 
364         res.render(page, {
365           userRole,
366           weather: weather,
367           place: place,
368           temp: weatherTemp,
369           icon: weatherIcon,
370           timezone: weatherTimezone,
371           fahrenheit: weatherFahrenheit,
372           main: main,
373           error: null,
374           treatItems,
375           entreeItems,
376           drinkItems,
377           sideItems,
378           saladItems,
379         });
380       }
381     })
382     .catch((err) => {
383       res.render(page, {
384         userRole,
385         treatItems,
386         entreeItems,
387         drinkItems,
388         sideItems,
389         saladItems,
390         weather: null,
391         error: "Error, please try again",
392       });
393     });
394 }
