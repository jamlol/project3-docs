320 const apiKey = `${process.env.WEATHER_API_KEY}`;
321 // Render the weather api for guest checkout
322 function guestRenderWeather(req, res, page) {
323   // Get city name passed in the form
324   let city = req.body.city;
325 
326   // Use that city name to fetch data
327   // Use the API_KEY in the '.env' file
328   let url = `http://api.openweathermap.org/data/2.5/weather?q=${city}&units=metric&appid=${apiKey}`;
329 
330   fetch(url)
331     .then((res) => res.json())
332     .then(async (weather) => {
333       const entree = new Menu();
334       const drink = new Menu();
335       const salad = new Menu();
336       const side = new Menu();
337       const treat = new Menu();
338 
339       const entreeItems = await entree.displayEntrees();
340       const drinkItems = await drink.displayDrinks();
341       const saladItems = await salad.displaySalads();
342       const sideItems = await side.displaySides();
343       const treatItems = await treat.displayTreats();
344       if (weather.main == undefined) {
345         res.render(page, {
346           userRole,
347           treatItems,
348           entreeItems,
349           drinkItems,
350           sideItems,
351           saladItems,
352           weather: null,
353           error: "Error, please try again",
354         });
355       } else {
356         let place = `${weather.name}, ${weather.sys.country}`,
357           weatherTimezone = `${new Date(
358             weather.dt * 1000 - weather.timezone * 1000
359           )}`;
360         let weatherTemp = `${weather.main.temp}`,
361           weatherIcon = `http://openweathermap.org/img/wn/${weather.weather[0].icon}@2x.png`,
362           main = `${weather.weather[0].main}`,
363           weatherFahrenheit;
364         weatherFahrenheit = (weatherTemp * 9) / 5 + 32;
365 
366         function roundToTwo(num) {
367           return +(Math.round(num + "e+2") + "e-2");
368         }
369         weatherFahrenheit = roundToTwo(weatherFahrenheit);
370 
371         userRole = userRoles[2].role;
372 
373         res.render(page, {
374           userRole,
375           weather: weather,
376           place: place,
377           temp: weatherTemp,
378           icon: weatherIcon,
379           timezone: weatherTimezone,
380           fahrenheit: weatherFahrenheit,
381           main: main,
382           error: null,
383           treatItems,
384           entreeItems,
385           drinkItems,
386           sideItems,
387           saladItems,
388         });
389       }
390     })
391     .catch((err) => {
392       res.render(page, {
393         userRole,
394         treatItems,
395         entreeItems,
396         drinkItems,
397         sideItems,
398         saladItems,
399         weather: null,
400         error: "Error, please try again",
401       });
402     });
403 }
